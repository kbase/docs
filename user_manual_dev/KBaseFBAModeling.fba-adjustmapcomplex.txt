fba-adjustmapcomplex <Map ID> <Complex ID> [-?ehvw] [long options...]
	--name                Name of complex
	--new                 Create new complex
	--delete              Delete specified complex
	--rolestoadd          Roles to add to complex (role
	                      id:triggering:optional:type)
	--rolestoremove       Roles to remove from complex (; delimited)
	-w --workspace        Workspace to save FBA results
	-e --showerror        Set as 1 to show any errors in execution
	-v --verbose          Print verbose messages
	-? -h --help          Print this usage information
