kbfba-adjustmodel <Model ID> <Reaction IDs (;-separated)> [-?acdeghioruvw] [long options...]
	-d --direction       Directionality for reaction (;-separated if
	                     multiple)
	-c --compartment     Compartment for reaction (;-separated if
	                     multiple)
	-i --compindex       Index of compartment for reaction (;-separated
	                     if multiple)
	-g --gpr             GPR for reaction (;-separated if multiple)
	-r --removerxn       Remove reaction(s)
	-a --addrxn          Add reaction(s)
	-w --workspace       Workspace to save FBA results
	-o --overwrite       Overwrite any existing FBA with same name
	-u --outputid        Output ID for adjusted reaction (by default,
	                     creates a new version of the same model)
	-e --showerror       Set as 1 to show any errors in execution
	-v --verbose         Print verbose messages
	-? -h --help         Print this usage information
