ga-seq-to-genome <Sequence ID> [-?cdehpstvw] [long options...]
	--genomeid              ID for created genome in workspace
	--seqws                 Workspace with sequence object
	-t --transcripts        Transcript sequence object
	-p --proteins           Protein sequence object
